# 函数
```

static int Main()
{
    var str = "oldstr";
    var oi = 100;
    outfun(1,out str,out oi);
    reffun(ref oi,ref str);
    return 0;
}

// 多返回值
public static void outfun(int b, out string str, out int a)
{
    str = "nba";
    a = b;
}

// 值传递变引用传递
public static void reffun(ref int a, ref string str)
{
    a = 10;
    str = "new str";
}

// 可变参数，永远放在最后边
public static double chparams(int a,params double[] vals)
{
    if (vals.Length == 0)
        return 0;
    
    var num = vals[0];
    foreach (var val in vals)
    {
        num = (num + val) / 2.0;
    }

    return num;
}


// 可选参数默认值必须可以在编译时确定，且不能和可变参数同时出现
public static void select(string name = "sb") {}

// 和可变参数搭配使用的是指定参数
select(a:2,b:1);

```
